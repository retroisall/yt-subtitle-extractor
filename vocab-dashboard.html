系統設定

連線設定

在 LINE Bot 輸入任意訊息後，可在 LINE 紀錄看到你的 userId

Firebase Firestore 安全規則（需手動貼入 Firebase Console）

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // 使用者單字庫（Chrome 套件，需登入）
    match /users/{uid}/words/{word} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }
    // LINE Bot 資料（Server 端寫入，Dashboard 需登入讀取）
    match /line_messages/{id}  { allow write: if true; allow read: if request.auth != null; }
    match /keywords/{id}       { allow write: if true; allow read: if request.auth != null; }
    match /schedule/{id}       { allow write: if true; allow read: if request.auth != null; }
    match /memory/{id}         { allow write: if true; allow read: if request.auth != null; }
    match /learn_words/{id}    { allow write: if true; allow read: if request.auth != null; }
    match /scores/{id}         { allow write: if true; allow read: if request.auth != null; }
    match /games/{id}          { allow write: if true; allow read: if request.auth != null; }
  }
}
